20. Type of Triangle

https://www.hackerrank.com/challenges/what-type-of-triangle/problem

SELECT CASE 
WHEN A + B > C THEN CASE 
    WHEN A = B AND B = C THEN 'Equilateral' 
    WHEN A = B OR B = C OR A = C THEN 'Isosceles' 
    WHEN A != B OR B != C OR A != C THEN 'Scalene' 
    END 
ELSE 'Not A Triangle' 
END 
FROM TRIANGLES;
